use rusqlite::{params, Connection, Result};
use std::path::PathBuf;
use std::sync::Mutex;
use::serde::Deserialize;

//Импорт типов для управления состоянием
use tauri::State;

//Описываем какие данные об игре мы передадим пользователю
//Serialize превратит структуру в JSON автоматически
#[derive(serde::Serialize)]
pub struct Game{
    pub id: String,
    pub name: String,
    pub install_path: Option<String>,
}

//Структура для парсинга входящего JSON
//Deserialize превращзщает JSON в эти структуры
#[derive(Deserialize)]
pub struct CatalogGame {
    pub id: String,
    pub name: String,
    pub localizations: Vec<CatalogLocalization>,
}

#[derive(Deserialize)]
pub struct CatalogLocalization {
    pub id: String,
    pub version: String,
    pub primary_url: String,
    pub backup_url: Option<String>,
    pub archive_hash: String,
    pub file_size_mb: i64,
    pub install_instructions: String, //JSON строка
    pub dll_whitelist: Option<String>, //JSON строка
}

#[tauri::command]
pub fn sync_catalog(state: State<DbState>, json_string: String) -> Result<(), String>{
    //Парсинг JSON, полученного от React
    let catalog: Vec<CatalogGame> = serde_json::from_str(&json_string).map_err(|e| format!("Ошибка парсинга JSON: {}", e))?;

    let mut conn = state.0.lock().map_err(|e| e.to_string())?;

    //Старт транзакции
    //Если на какой то игре произойдет ошибка транзакция откатит всё назад и в бд не останются полу-данные

    let tx = conn.transaction().map_err(|e| e.to_string())?;

    for game in catalog {
        //Добавляем или обвновляем игру
        tx.execute(
            "INSERT INTO games (id, game) VALUES (?1, ?2)
            ON CONFLICT(id) DO UPDATE SET name = excluded.name",
            params![game.id, game.name],
        ).map_err(|e| e.to_string())?;

        for loc in game.localizations {
            tx.execute(
                "INSERT INTO localizations (id, game_id, version, primary_url, backup_url, archive_hash, file_size_mb, install_instructions, dll_whitelist)
                VALUES (?1, ?2, ?3, ?4, ?5, ?6, ?7, ?8, ?9)
                ON CONFLICT(id) DO UPDATE SET 
                version=excluded.version, primary_url=excluded.primary_url, 
                backup_url=excluded.backup_url, archive_hash=excluded.archive_hash,
                file_size_mb=excluded.file_size_mb, install_instructions=excluded.install_instructions,
                dll_whitelist=excluded.dll_whitelist",
                params![
                    loc.id, 
                    game.id, // <-- Важно: привязываем к ID игры
                    loc.version, 
                    loc.primary_url, 
                    loc.backup_url, 
                    loc.archive_hash, 
                    loc.file_size_mb, 
                    loc.install_instructions, 
                    loc.dll_whitelist
                ],
            ).map_err(|e| e.to_string())?;
        }
    }

    //Подтверждение транзации и отправка файлов на бд
    tx.commit().map_err(|e| e.to_string())?;

    Ok(())
}

//Обёратка для хранения в памяти нашего соединения
pub struct DbState(pub Mutex<Connection>);
//Инициализация бд
pub fn init(app_data_dir: PathBuf) -> Result<Connection>{
    //Проверка на существование папки данных приложения
    if !app_data_dir.exists() {
        std::fs::create_dir_all(&app_data_dir)
            .expect("Не удалось создать папку данных приложения");
    }

    //Формируем путь к файлу бд
    let db_path = app_data_dir.join("launcher.db");

    //Открываем или создаём файл бд
    let conn = Connection::open(db_path)?;

    //Настройки производительности и безопасности
    //WAL -> позвоялет читать и одновременно записывать данные в бд
    //Foreign Keys -> Включение поддержки внешних ключей

    conn.execute_batch(
        "PRAGMA journal_mode = WAL;
        PRAGMA foreign_keys = ON;"
    )?;

    //Создание таблиц
    conn.execute_batch(
        "
        CREATE TABLE IF NOT EXISTS games (
            id TEXT PRIMARY KEY,
            name TEXT NOT NULL,
            install_path TEXT
        );

        CREATE TABLE IF NOT EXISTS localizations (
            id TEXT PRIMARY KEY,
            game_id TEXT NOT NULL,
            version TEXT NOT NULL,
            primary_url TEXT NOT NULL,
            backup_url TEXT,
            archive_hash TEXT NOT NULL,
            file_size_mb INTEGER,
            install_instructions TEXT NOT NULL,
            dll_whitelist TEXT,
            FOREIGN KEY (game_id) REFERENCES games(id) ON DELETE CASCADE
        
        );

        CREATE TABLE IF NOT EXISTS install_states (
            localization_id TEXT PRIMARY KEY,
            status TEXT NOT NULL DEFAULT 'available',
            installed_version TEXT,
            backup_path TEXT,
            error_message TEXT,
            FOREIGN KEY (localization_id) REFERENCES localizations(id) ON DELETE CASCADE
        );
        "
    )?;

    Ok(conn)
}

/*Если удалить игру из таблицы, то автоматически удалятся
все её локализации из localizations и все статусы установок из install_states

launcher.db хранится в AppData
*/

#[tauri::command]
pub fn get_games(state: State<DbState>) -> Result<Vec<Game>, String> {
    //Блокировка  бд для чтения.
    let conn = state.0.lock().map_err(|e| e.to_string())?;

    //Подготовка SQL запроса
    let mut stmt = conn
        .prepare("SELECT id, name, install_path FROM games")
        .map_err(|e| e.to_string())?;
    
    //Выполнение запроска и превращение строк БД в структуру Game
    let games_iter = stmt
        .query_map([], |row| {
            Ok(Game {
                id: row.get(0)?,
                name: row.get(1)?,
                install_path: row.get(2)?,
            })
        })
        .map_err(|e| e.to_string())?;

    let mut games = Vec::new();
    for game in games_iter {
        games.push(game.map_err(|e| e.to_string())?);
    }

    //Возвращаем список фо фронт, tauri сам преобразует его в JSON
    Ok(games)
}
